    <script type="importmap">
        {
            "imports": {
                "three": "https://unpkg.com/three@0.160.0/build/three.module.js",
                "three/addons/": "https://unpkg.com/three@0.160.0/examples/jsm/",

    <script type="module">
        import * as THREE from 'three';
        import { OrbitControls } from 'three/addons/controls/OrbitControls.js';
        import GUI from 'lil-gui';
        // --- Configuration ---
        const CONFIG = {
            particleCount: 256, // 256x256 = 65,536 particles
            pointSize: 2.0,
            bgColor: 0x050505,
            defaultImage: 'https://images.unsplash.com/photo-1543783256-4b6c23cb365b?q=80&w=2000&auto=format&fit=crop' // 默认占位图 (瓷器/纹理)
        };
        const DATA_LIST = [
            { cn: "1", en: "MARTIAL ARTS" },
            { cn: "2", en: "TCM" },
            { cn: "3", en: "PEKING OPERA" },
            { cn: "4", en: "TEA CEREMONY" },
            { cn: "5", en: "PORCELAIN" },
            { cn: "6", en: "HANFU" },
            { cn: "7", en: "GO" },
            { cn: "8", en: "PAPER CUTTING" },
            { cn: "9", en: "EMBROIDERY" }
        ];
        // --- Global Variables ---
        let scene, camera, renderer, controls, particleSystem;
        let uniforms;
        const clock = new THREE.Clock();
        // --- Shader Definitions (GLSL) ---
        // Vertex Shader: 处理位置、呼吸、浮雕、爆炸
        const vertexShader = `
            uniform float uTime;
            uniform float uRelief;      // 3D浮雕强度
            uniform float uSize;        // 粒子大小
            uniform sampler2D uTexture; // 图像纹理
            // 动态控制
            uniform float uHeartbeatAmp;
            uniform float uHeartbeatSpeed;
            uniform float uFluidStrength;
            varying vec2 vUv;
            varying float vLuma;
            // 简单的随机函数
            float random(vec2 st) {
                return fract(sin(dot(st.xy, vec2(12.9898,78.233))) * 43758.5453123);
            }
            void main() {
                vUv = uv;
                // 采样颜色计算亮度
                vec4 texColor = texture2D(uTexture, uv);
                float luma = dot(texColor.rgb, vec3(0.299, 0.587, 0.114));
                vLuma = luma;
                vec3 pos = position;
                // 1. 3D Relief (基于亮度沿Z轴位移)
                pos.z += luma * uRelief;
                // 2. Fluid / Scatter (基于噪波的流动效果)
                float noise = sin(pos.y * 10.0 + uTime * 2.0) * cos(pos.x * 10.0 + uTime);
                pos.x += noise * uFluidStrength * (1.0 - luma); // 暗部动得更多
                pos.y += noise * uFluidStrength * 0.5;
                // 3. Heartbeat (整体呼吸)
                float breath = 1.0 + sin(uTime * uHeartbeatSpeed) * uHeartbeatAmp;
                pos *= breath;
                vec4 mvPosition = modelViewMatrix * vec4(pos, 1.0);
                // 粒子大小随距离衰减
                gl_PointSize = uSize * (300.0 / -mvPosition.z);
                gl_Position = projectionMatrix * mvPosition;
            }
        `;
        // Fragment Shader: 处理颜色、抠图、调色
        const fragmentShader = `
            uniform sampler2D uTexture;
            uniform float uThreshold;   // 去黑阈值
            // 调色板
            uniform float uBrightness;
            uniform float uContrast;
            uniform float uTemperature; // 色温 (-1 blue to 1 orange)
            uniform float uTint;        // 色调 (-1 green to 1 magenta)
            varying vec2 vUv;
            varying float vLuma;
            void main() {
                vec4 color = texture2D(uTexture, vUv);
                // 1. Smart Luma Key (去黑)
                if (vLuma < uThreshold) {
                    discard;
                }
                // 2. 圆形粒子裁切 (可选，让粒子变成圆点)
                vec2 coord = gl_PointCoord - vec2(0.5);
                if(length(coord) > 0.5) discard;
                // 3. Color Grading Pipeline
                vec3 rgb = color.rgb;
                // Contrast
                rgb = (rgb - 0.5) * uContrast + 0.5;
                // Brightness
                rgb = rgb + uBrightness;
                // Temperature (简易算法: R增加B减少)
                rgb.r += uTemperature * 0.1;
                rgb.b -= uTemperature * 0.1;
                // Tint (G减少)
                rgb.g -= uTint * 0.1;
                // 增强金色感 (Cyberpunk Gold Overlay)
                // 基于亮度的着色，高光偏金
                vec3 gold = vec3(0.83, 0.68, 0.21);
                rgb = mix(rgb, gold, vLuma * 0.2);
                gl_FragColor = vec4(rgb, 1.0);
            }
        `;
        // --- Initialization ---
        function init() {
            // Scene Setup
            scene = new THREE.Scene();
            scene.background = new THREE.Color(CONFIG.bgColor);
            // 简单的环境雾
            scene.fog = new THREE.FogExp2(CONFIG.bgColor, 0.002);
            // Camera
            camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 1000);
            camera.position.set(0, 0, 180);
            // Renderer
            renderer = new THREE.WebGLRenderer({ antialias: true, alpha: false });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2)); // 性能优化
            document.getElementById('canvas-container').appendChild(renderer.domElement);
            // Controls (Orbit Cruise)
            controls = new OrbitControls(camera, renderer.domElement);
            controls.enableDamping = true;
            controls.dampingFactor = 0.05;
            controls.autoRotate = true;
            controls.autoRotateSpeed = 0.5;
            controls.enablePan = false;
            controls.minDistance = 50;
            controls.maxDistance = 300;
            // Build Particles
            createParticleSystem();
            // Build UI
            buildList();
            setupGUI();
            setupInteraction();
            // Remove Loader
            setTimeout(() => {
                document.getElementById('loader').style.opacity = 0;
                setTimeout(() => document.getElementById('loader').remove(), 500);
            }, 1000);
            // Start Loop
            animate();
        }
        function createParticleSystem() {
            const count = CONFIG.particleCount; // 256
            const numParticles = count * count;
            const geometry = new THREE.BufferGeometry();
            const positions = new Float32Array(numParticles * 3);
            const uvs = new Float32Array(numParticles * 2);
            // 生成网格
            let i = 0;
            for (let y = 0; y < count; y++) {
                for (let x = 0; x < count; x++) {
                    // Normalize -0.5 to 0.5
                    const u = x / count;
                    const v = y / count;
                    // Centered grid
                    const px = (u - 0.5) * 200; // 宽度缩放
                    const py = (v - 0.5) * 200; // 高度缩放
                    const pz = 0;
                    positions[i * 3] = px;
                    positions[i * 3 + 1] = py;
                    positions[i * 3 + 2] = pz;
                    uvs[i * 2] = u;
                    uvs[i * 2 + 1] = v;
                    i++;
                }
            }
            geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
            geometry.setAttribute('uv', new THREE.BufferAttribute(uvs, 2));
            // Load Initial Texture
            const textureLoader = new THREE.TextureLoader();
            const initialTexture = textureLoader.load(CONFIG.defaultImage);
            // Uniforms setup
            uniforms = {
                uTime: { value: 0 },
                uTexture: { value: initialTexture },
                uSize: { value: CONFIG.pointSize },
                // Visual / Keying
                uThreshold: { value: 0.1 },
                uRelief: { value: 30.0 },
                // Color
                uBrightness: { value: 0.0 },
                uContrast: { value: 1.1 },
                uTemperature: { value: 0.0 },
                uTint: { value: 0.0 },
                // Dynamics
                uHeartbeatAmp: { value: 0.0 },
                uHeartbeatSpeed: { value: 2.0 },
                uFluidStrength: { value: 0.0 }
            };
            const material = new THREE.ShaderMaterial({
                uniforms: uniforms,
                vertexShader: vertexShader,
                fragmentShader: fragmentShader,
                transparent: true,
                depthWrite: false,
                blending: THREE.AdditiveBlending // 赛博朋克发光感
            });
            particleSystem = new THREE.Points(geometry, material);
            scene.add(particleSystem);
        }
        function buildList() {
            const listContainer = document.getElementById('content-list');
            DATA_LIST.forEach(item => {
                const el = document.createElement('div');
                el.className = 'list-item';
                el.innerHTML = `
                    <span class="item-en">${item.en}</span>
                    <span>${item.cn}</span>
                `;
                el.onclick = () => {
                    // 模拟切换内容的视觉反馈
                    document.querySelectorAll('.list-item').forEach(d => d.classList.remove('active'));
                    el.classList.add('active');
                    console.log(`Switching to category: ${item.cn}`);
                    // 此处在实际项目中会请求后端获取该分类的图，这里仅做交互演示
                    triggerGlitchEffect();
                };
                listContainer.appendChild(el);
            });
        }
        function triggerGlitchEffect() {
            // 简单的切换故障效果
            const originalAmp = uniforms.uFluidStrength.value;
            uniforms.uFluidStrength.value = 5.0; // 瞬间爆炸
            setTimeout(() => {
                const interval = setInterval(() => {
                    uniforms.uFluidStrength.value *= 0.9;
                    if(uniforms.uFluidStrength.value <= originalAmp + 0.1) {
                        uniforms.uFluidStrength.value = originalAmp;
                        clearInterval(interval);
                    }
                }, 16);
            }, 100);
        }
        function setupGUI() {
            const gui = new GUI({ title: 'CONTROL PANEL' });
            gui.domElement.style.marginTop = "60px"; // 避开标题
            // 视觉控制 (Visual)
            const folderVisual = gui.addFolder('Visual & Color');
            folderVisual.add(uniforms.uSize, 'value', 0.1, 10.0).name('Particle Size');
            folderVisual.add(uniforms.uThreshold, 'value', 0.0, 0.5).name('Luma Cutoff'); // 去黑阈值
            folderVisual.add(uniforms.uRelief, 'value', 0.0, 100.0).name('3D Relief'); // 浮雕
            // 调色板 (Palette)
            folderVisual.add(uniforms.uBrightness, 'value', -0.5, 0.5).name('Brightness');
            folderVisual.add(uniforms.uContrast, 'value', 0.5, 2.0).name('Contrast');
            folderVisual.add(uniforms.uTemperature, 'value', -1.0, 1.0).name('Temp (K)');
            folderVisual.add(uniforms.uTint, 'value', -1.0, 1.0).name('Tint');
            // 动态 (Dynamics)
            const folderDynamics = gui.addFolder('Dynamics');
            folderDynamics.add(uniforms.uHeartbeatAmp, 'value', 0.0, 0.5).name('Breath Amp');
            folderDynamics.add(uniforms.uHeartbeatSpeed, 'value', 0.1, 10.0).name('Breath Spd');
            folderDynamics.add(uniforms.uFluidStrength, 'value', 0.0, 10.0).name('Fluid Chaos');
            // 摄像机
            const folderCam = gui.addFolder('Camera');
            folderCam.add(controls, 'autoRotate').name('Orbit Cruise');
            folderCam.add(controls, 'autoRotateSpeed', 0.1, 5.0).name('Cruise Speed');
            gui.close(); // 默认折叠，保持界面整洁
        }
        function setupInteraction() {
            // 文件上传逻辑
            const fileInput = document.getElementById('file-input');
            fileInput.addEventListener('change', (e) => {
                const file = e.target.files[0];
                if (!file) return;
                const reader = new FileReader();
                reader.onload = (event) => {
                    const img = new Image();
                    img.src = event.target.result;
                    img.onload = () => {
                        // 更新纹理
                        const newTexture = new THREE.Texture(img);
                        newTexture.needsUpdate = true;
                        uniforms.uTexture.value = newTexture;
                        triggerGlitchEffect(); // 触发切换特效
                    };
                };
                reader.readAsDataURL(file);
            });
            // 窗口大小调整
            window.addEventListener('resize', onWindowResize, false);
        }
        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }
        function animate() {
            requestAnimationFrame(animate);
            const elapsedTime = clock.getElapsedTime();
            uniforms.uTime.value = elapsedTime;
            controls.update();
            renderer.render(scene, camera);
        }
        // Run
        init();
    </script>
